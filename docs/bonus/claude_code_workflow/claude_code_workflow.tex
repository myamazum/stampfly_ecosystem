% === P2: コーディングエージェントとは ===
\begin{frame}{コーディングエージェントとは}
  \begin{block}{定義}
    AI がターミナル（黒い画面のコマンド入力画面）上でコードを読み書きし、
    ビルド・テスト・変更履歴の記録まで\textbf{自律的に実行}するツール
  \end{block}

  \vspace{0.3em}

  \begin{table}
    \centering\small
    \begin{tabular}{lll}
      \hline
      & \textbf{チャット AI} & \textbf{コーディングエージェント} \\
      \hline
      動作     & 対話（コピペが必要）  & ファイルを直接編集 \\
      範囲     & 1ファイルの断片      & プロジェクト全体 \\
      実行     & ユーザーが手動実行    & ビルド・テストも自律実行 \\
      履歴管理 & 関与しない           & 変更の記録・共有まで \\
      \hline
    \end{tabular}
  \end{table}

  \vspace{0.3em}

  {\footnotesize \textbf{なぜターミナルで動くのか:}
    エディタに依存しないので、\textbf{どんな開発環境でも同じように使える}}
\end{frame}

% ------------------------------------------------------------------
% === P3: Claude Code とは ===
\begin{frame}{Claude Code とは}
  \begin{block}{概要}
    {\small Anthropic 社が提供するコーディングエージェント。
    ターミナルから Claude を呼び出し、コードの読み書きからテスト・変更記録まで自律的に実行する}
  \end{block}

  \begin{table}
    \centering\small
    \begin{tabular}{ll}
      \hline
      \textbf{機能} & \textbf{説明} \\
      \hline
      指示書ファイル    & プロジェクトのルールを書いたファイルを毎回読む \\
      記憶の永続化      & 会話で学んだことを次回以降も覚えている \\
      計画モード        & コードを書かずに調査・設計だけさせるモード \\
      Git 連携          & 変更の記録・共有を自動で行う \\
      自動化の仕組み    & 繰り返す作業やチェックを自動実行 \\
      \hline
    \end{tabular}
  \end{table}

  {\scriptsize
    ※ 各機能の詳細は後半で解説 \quad
    インストール: \texttt{npm install -g @anthropic-ai/claude-code} \quad
    起動: \texttt{cd your-project \&\& claude}
  }
\end{frame}

% === P4: まず「何を作るか」を決める ===
\begin{frame}{まず「何を作るか」を決める — 要件定義}
  \begin{alertblock}{AI に「いい感じに作って」は通じない}
    {\footnotesize
    曖昧な指示 → AI が勝手に解釈 → 意図と違うものができる。\\
    \warn{最初に「何を作るか」を具体的に言葉にする}ことが最も重要}
  \end{alertblock}

  \begin{exampleblock}{要件を具体化する3ステップ}
    {\footnotesize
    \textbf{1. 目的:} このプログラムは何のためにあるか？（1文で）\\
    \textbf{2. 機能:} 具体的に何ができればいいか？（箇条書き）\\
    \textbf{3. 制約:} 使う言語・ライブラリ・動作環境は？}
  \end{exampleblock}

  \begin{block}{例: TODO アプリ}
    {\scriptsize
    \textbf{目的:} コマンドラインでタスクを管理する \quad
    \textbf{機能:} 追加・一覧・完了・削除 \quad
    \textbf{制約:} Python、外部ライブラリなし}
  \end{block}

  {\scriptsize \textbf{Tips:} 要件が具体的なほど AI の出力品質が上がる。チュートリアルで実践します}
\end{frame}

% ------------------------------------------------------------------
% === P5: なぜ「計画」が必要か ===
\begin{frame}{なぜ「計画」が必要か}
  \begin{alertblock}{AI に自由にコードを書かせると何が起きるか}
    \begin{itemize}\small
      \item 意図と違う設計で大量のファイルを変更 → \warn{手戻りコスト大}
      \item テストなしで実装を進める → バグの発見が遅れる
      \item 既存コードの意図を誤解 → 動いていた機能が壊れる
    \end{itemize}
  \end{alertblock}

  \vspace{0.5em}

  \begin{exampleblock}{解決策: 「承認」ステップを挟む}
    {\small
    \textbf{1.} AI に計画を出させる（コードは書かせない）\\
    \textbf{2.} 人間が計画をレビューして承認/修正 \\
    \textbf{3.} 承認後に初めて実装を開始
    }
  \end{exampleblock}

  \vspace{0.3em}

  {\footnotesize \textbf{原則:} 「考える」と「書く」を分離する。これが手戻りを防ぐ最大のコツ}
\end{frame}

% ------------------------------------------------------------------
% === P5: Plan Mode — 設計してから書く ===
\begin{frame}{Plan Mode — 設計してから書く}
  \begin{block}{Plan Mode（計画モード）とは}
    {\small \textbf{Shift+Tab} キーで切り替え。コード編集を\textbf{できない}読み取り専用モード。
    AI に調査・設計だけをさせ、方針が固まってから実装に移る}
  \end{block}

  \begin{columns}[T]
    \begin{column}{0.45\textwidth}
      \begin{alertblock}{有効な場面}
        \begin{itemize}\footnotesize
          \item 複数ファイルにまたがる変更
          \item 設計の判断が必要なとき
          \item 失敗コストが高い作業
          \item 初めて触るプロジェクト
        \end{itemize}
      \end{alertblock}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{exampleblock}{Plan Mode でできること}
        {\footnotesize
        ファイルを読んで調査する \\
        プロジェクトの構造を分析する \\
        変更計画を作って提案する \\[3pt]
        \warn{$\times$} ファイルの編集・作成 \\
        \warn{$\times$} コマンドの実行
        }
      \end{exampleblock}
    \end{column}
  \end{columns}

  {\scriptsize \textbf{Tips:} 大きな変更ほど計画モードが効く。5分の計画で1時間の手戻りを防げる}
\end{frame}

\begin{frame}{Plan → Approve → Implement サイクル}
  \begin{center}

      \node[box, below=of impl]  (test)    {ビルド/テスト\\で検証};
      \node[box, below=of approve] (commit) {変更を\\記録};
      \node[box, below=of plan]   (next)   {次のタスク\\or 完了};

      \draw[arr] (plan)    -- (approve);
      \draw[arr] (approve) -- (impl);
      \draw[arr] (impl)    -- (test);
      \draw[arr] (test)    -- (commit);
      \draw[arr] (commit)  -- (next);

      % Reject loop
      \draw[arr, sfred] (approve.north) -- ++(0,6mm)
        -| (plan.north)
        node[above, font=\scriptsize, sfred, pos=0.3] {修正依頼};

      % Fix loop
      \draw[arr, sforange] (test.east) -- ++(8mm,0) |- (impl.east)
        node[right, font=\scriptsize, sforange, pos=0.25] {失敗};
    \end{tikzpicture}
  \end{center}

  \vspace{0.2em}

  {\scriptsize
    \textcolor{sfred}{\textbf{赤枠}} = 人間の承認ポイント。
    ここで方向修正できるから安全に AI を使える。\\
    \textbf{Tips:} 大きな変更はフェーズに分割して計画 → 各フェーズを独立してコミット・検証}
\end{frame}

% ------------------------------------------------------------------
% === P7: Plan Mode の実践例 ===
\begin{frame}[fragile]{Plan Mode の実践例}
  {\small \textbf{例:} 「CSV を読み込んでグラフ表示する機能を追加して」→ 計画モードで出る計画:}

  \begin{lstlisting}[language={}, numbers=none]
## Plan: CSV 読み込み + グラフ表示機能

### 変更対象ファイル
1. src/csv_reader.py   (新規) - CSV 読み込み処理
2. src/plot.py         (新規) - グラフ描画処理
3. src/main.py         (既存) - メニューに項目追加
4. tests/test_csv.py   (新規) - 読み込みのテスト

### 設計判断
- CSV ライブラリ: 標準の csv モジュールを使用
- グラフ描画: matplotlib を使用
- ファイル選択: コマンドライン引数で指定

### 影響範囲
- 既存の機能には影響なし
- requirements.txt に matplotlib を追加
  \end{lstlisting}

  {\scriptsize
    ユーザーが「3ファイルに分けず1ファイルにまとめて」→ 計画を修正 → 承認 → 実装開始 \\
    \textbf{Tips:} 「バグを直して」より「バグが起きない設計にして」と計画させる方が効果的}
\end{frame}

% === P8: Git とは — 歴史と価値 ===
\begin{frame}{Git とは — 歴史と「なぜ世界標準になったか」}
  \begin{block}{Git の誕生}
    {\small
    \textbf{2005年}、Linux の生みの親 Linus Torvalds が Linux カーネル開発のために作成。\\
    数千人が同時に開発する巨大プロジェクトを管理するために生まれた}
  \end{block}

  \vspace{0.3em}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Git が解決した問題}
        \begin{itemize}\footnotesize
          \item 「誰が・いつ・何を・なぜ」変えたか完全に記録
          \item いつでも過去の状態に戻せる
          \item 複数人が同時に作業できる
          \item 変更を安全に統合できる
        \end{itemize}
      \end{exampleblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{しかし使い方が難しい}
        \begin{itemize}\footnotesize
          \item コマンドが多く覚えきれない
          \item 操作を間違えると怖い
          \item 初心者の最大の挫折ポイント
        \end{itemize}
        \vspace{0.3em}
        {\footnotesize → \textbf{便利さは圧倒的だが\\ハードルが高い}}
      \end{alertblock}
    \end{column}
  \end{columns}

  \vspace{0.3em}

  {\footnotesize 現在、GitHub 上のプロジェクト数は \textbf{3億以上}。Git はソフトウェア開発の事実上の標準}
\end{frame}

% === P9: AI × Git — 難しさを AI に任せ、旨みだけを得る ===
\begin{frame}{AI × Git — 難しさを AI に任せ、旨みだけを得る}
  \begin{block}{AI コーディングエージェントが Git を操作する}
    {\small
    Git の複雑なコマンドは\textbf{全て AI が代行}。
    あなたは「記録して」「戻して」と言うだけ}
  \end{block}

  \vspace{0.3em}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{AI の弱点を Git が補う}
        \begin{itemize}\footnotesize
          \item AI は会話が長くなると\textbf{忘れる}
          \item Git ログ = AI の\textbf{外部記憶装置}
          \item 「前回何をしたか」を Git から復元
          \item Next steps で再開ポイントを記録
        \end{itemize}
      \end{exampleblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Git ログの副産物}
        \begin{itemize}\footnotesize
          \item \textbf{作業記録}が自動的に残る
          \item \textbf{作業日報}を git log から生成可能
          \item 「何時間で何をしたか」が一目瞭然
          \item コードレビューの材料にもなる
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.3em}

  {\footnotesize \warn{Git なしで AI にコードを書かせるのは、セーブなしでゲームをするようなもの}。使わない手はない}
\end{frame}

% === P11: コミット戦略 ===
\begin{frame}[fragile]{コミット戦略}
  \begin{block}{コミット（変更の記録）のルールを決めておく}
    {\scriptsize 「何をなぜ変えたか」が伝わる記録を残す。Claude Code が自動でこの形式に従う:}
    \begin{lstlisting}[language={}, numbers=none]
feat(csv): add CSV import and graph display
                     ← 種類(範囲): 変更の要約（1行）
Changes:
- CSV 読み込み処理を実装     ← 何を変えたか
- matplotlib でグラフ描画機能を追加

Next steps:                  ← 次にやるべきこと
- tests/test_csv.py でテストを実行
- 大きな CSV ファイルでの動作確認
    \end{lstlisting}
  \end{block}

  \begin{columns}[T]
    \begin{column}{0.45\textwidth}
      {\small \textbf{記録するタイミング:}}
      \begin{itemize}\scriptsize
        \item 1機能 = 1記録（小さな単位で）
        \item テスト成功後
        \item 作業の区切り・中断時
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      {\small \textbf{Next steps（次の作業）の効果:}}
      \begin{itemize}\scriptsize
        \item 次回起動時の再開ポイントになる
        \item \textbf{コマンドレベル}で具体的に書く
        \item AI が「前回の続き」を即座に把握
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

% === P12: GitHub 連携 — PR とレビュー ===
\begin{frame}{GitHub 連携 — 変更をチームに共有する}
  \begin{block}{Pull Request（PR）= 「この変更をレビューしてください」という依頼}
    {\small
    \texttt{「この変更で PR を作成して」} と Claude Code に指示するだけで:}
    \begin{enumerate}\small
      \item 変更内容を分析して説明文を自動生成
      \item GitHub 上に PR（変更のレビュー依頼）を作成
      \item チームメンバーのレビューコメントへの対応も可能
    \end{enumerate}
  \end{block}

  \vspace{0.2em}

  \begin{center}
    \begin{tikzpicture}[
      node distance=5mm and 10mm,
      box/.style={rectangle, rounded corners=3pt, draw=sfblue,
                  fill=sflight, minimum height=7mm, minimum width=22mm,
                  font=\scriptsize, align=center},
      arr/.style={-{Stealth[length=2mm]}, sfblue, thick},
    ]
      \node[box] (branch) {ブランチを\\作成};
      \node[box, right=of branch] (impl) {実装 +\\変更を記録};
      \node[box, right=of impl] (pr) {PR を作成\\（レビュー依頼）};
      \node[box, right=of pr] (review) {チームが\\確認};
      \node[box, right=of review] (merge) {承認 →\\mainに統合};

\begin{frame}{安全な運用 — main ブランチを直接触らない}
  \begin{alertblock}{main（main）を直接変更しない}
    {\small Git では「main」と「ブランチ」を分けられる。AI にmainを直接変更させると、
    レビューなしで反映されてしまう。必ずブランチ → レビュー → 統合の流れを守る}
  \end{alertblock}

  \vspace{0.2em}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{推奨フロー}
        {\scriptsize
        \textbf{1.} ブランチを作成 \\
        \textbf{2.} その枝で実装・テスト・記録 \\
        \textbf{3.} レビュー依頼を作成 \\
        \textbf{4.} 確認・承認後にmainへ統合 \\
        \textbf{5.} ブランチを削除
        }
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{AI にやらせないことを決める}
        {\scriptsize
        設定ファイルで制御:
        \begin{itemize}
          \item mainへの直接書き込みをブロック
          \item 使えるコマンドを制限
          \item GitHub 側でもmainを保護できる
        \end{itemize}
        }
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.2em}

  {\footnotesize \textbf{原則:} AI に「何をさせるか」だけでなく「何をさせないか」も設定する}
\end{frame}

% === P22: 初心者がやりがちな失敗 ===
\begin{frame}{初心者がやりがちな失敗}
  \begin{table}
    \centering\footnotesize
    \begin{tabular}{p{40mm}p{55mm}}
      \hline
      \textbf{失敗パターン} & \textbf{何が起きるか} \\

        意図と違う設計で大量変更 → やり直し \\
      変更履歴を管理しない &
        AI の変更を追えない → 戻せない \\
      指示書ファイルを書かない &
        毎回同じ指示を繰り返す → 非効率 \\
      mainに直接変更を送る &
        レビューなしで本番反映 → 障害リスク \\
      自動化を使わず手作業を繰り返す &
        書式チェック忘れ・整形崩れが頻発 \\
      記憶機能を活用しない &
        同じミスを何度も指摘する必要がある \\
      \hline
    \end{tabular}
  \end{table}

  {\scriptsize \textbf{共通の原因:}
    AI を「魔法の箱」として扱ってしまう →
    \textbf{対策:} 安全装置（計画・変更管理・指示書）を先に整える \\
    \textbf{目安:} fix が3回以上続いたら手を止めて Plan Mode に戻る（場当たり修正の連鎖を断つ）}
\end{frame}

\begin{frame}{Top 10 Tips}

    \item \textbf{Claude Code を育てる} — 失敗するたびにルールを追加
    \item \textbf{フック（Hooks）で自動化} — 手作業の繰り返しを排除
    \item \textbf{スキル（Skills）で1コマンドに} — 複雑な手順をまとめる
    \item \textbf{GitHub = AI の知識の共有拠点} — クローンすれば知識も一緒に来る

\begin{frame}{チュートリアル: TODO アプリを作ろう}
  \begin{block}{概要}
    {\small Claude Code のワークフローを体験する実践演習。\\
    コマンドライン TODO アプリ（Python）を\textbf{ゼロから完成}させる}
  \end{block}

  \vspace{0.3em}

  \begin{table}
    \centering\footnotesize
    \begin{tabular}{rlp{60mm}}
      \hline
      \textbf{Step} & \textbf{内容} & \textbf{学ぶこと} \\
      \hline
      1 & インストールと初期設定 & 環境構築、CLAUDE.md 作成 \\
      2 & 要件を定義する & 目的・機能・制約の言語化 \\
      3 & 計画を立てる & Plan Mode の使い方 \\
      4 & 実装する & 計画に基づく段階的な実装 \\
      5 & バグを見つけて直す & テスト、デバッグ、コミット \\
      6 & 振り返り & CLAUDE.md の改善サイクル \\
      \hline
    \end{tabular}
  \end{table}

  {\scriptsize \textbf{前提:} Python 3 と Node.js がインストール済みであること}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{Step 1: インストールと初期設定}
  \begin{block}{Claude Code をインストールして、プロジェクトを準備する}
    \begin{lstlisting}[language={}, numbers=none]
# 1. Claude Code をインストール
npm install -g @anthropic-ai/claude-code
# 2. プロジェクトフォルダを作成
mkdir todo-app && cd todo-app
# 3. Git リポジトリを初期化
git init
# 4. Claude Code を起動して CLAUDE.md を生成
claude /init
    \end{lstlisting}
  \end{block}

  \begin{exampleblock}{CLAUDE.md を編集して最初のルールを書く}
    \begin{lstlisting}[language={}, numbers=none]
## Build
python todo.py
## Rules
- Python 3、外部ライブラリなし
- データは todo.json に保存
    \end{lstlisting}
  \end{exampleblock}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{Step 2: 要件を Claude Code に伝える}
  \begin{block}{目的・機能・制約を具体的に伝える}
    {\small Claude Code に以下のように入力する:}
  \end{block}

  \begin{lstlisting}[language={}, numbers=none]
コマンドライン TODO アプリを作りたい。

目的: ターミナルでタスクを管理する
機能:
- タスクの追加 (add "タスク名")
- 一覧表示 (list) - 未完了/完了を区別
- 完了にする (done 番号)
- 削除 (delete 番号)
制約:
- Python 3、外部ライブラリなし
- データは todo.json に保存
- 日本語対応

まず Plan Mode で設計してください。
  \end{lstlisting}

  {\scriptsize \textbf{ポイント:} 「まず Plan Mode で」と指示することで、いきなりコードを書かせない}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{Step 3: 計画を確認して承認する}
  \begin{block}{Claude Code が計画を出す → 人間が確認}
    {\small 計画モードで以下のような出力が返ってくる:}
  \end{block}

  \begin{lstlisting}[language={}, numbers=none]
## Plan: コマンドライン TODO アプリ

### ファイル構成
1. todo.py     - メイン処理(引数解析、CRUD操作)
2. todo.json   - データ保存(自動生成)

### 設計
- argparse で add/list/done/delete を処理
- JSON ファイルでデータ永続化
- 各タスクは {id, text, done} の辞書
  \end{lstlisting}

  \vspace{0.2em}

  {\small
  \textbf{確認ポイント:} ファイル構成は妥当か？ 設計は要件を満たすか？ \\
  問題なければ「OK、実装して」と承認。修正したければ「ファイルは1つにして」等}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{Step 4: 実装してテストする}
  \begin{block}{承認後、Claude Code が実装を開始する}
    {\small 実装が完了したら、自分で動かして確認:}
  \end{block}

  \begin{lstlisting}[language={}, numbers=none]
# タスクを追加
python todo.py add "牛乳を買う"
python todo.py add "レポートを書く"

# 一覧表示
python todo.py list
  [ ] 1: 牛乳を買う
  [ ] 2: レポートを書く

# 完了にする
python todo.py done 1

# 再度一覧 → 完了マークがつく
python todo.py list
  [x] 1: 牛乳を買う
  [ ] 2: レポートを書く
  \end{lstlisting}

  {\scriptsize \textbf{動いたらコミット:} 「ここまでコミットして」→ Claude Code が自動で記録}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{Step 5: バグを見つけて修正する}
  \begin{alertblock}{意図的にバグを探してみる}
    {\small 以下を試して、おかしな動作がないか確認:}
  \end{alertblock}

  \begin{lstlisting}[language={}, numbers=none]
# 存在しない番号を完了にしたら？
python todo.py done 999

# 引数なしで実行したら？
python todo.py

# 空文字のタスクを追加したら？
python todo.py add ""
  \end{lstlisting}

  \begin{exampleblock}{バグを Claude Code に修正させる}
    {\small Claude Code に伝える:}

    \begin{lstlisting}[language={}, numbers=none]
done 999 でエラーになる。存在しない番号には
エラーメッセージを表示してください。
引数なしの場合はヘルプを表示してください。
    \end{lstlisting}
  \end{exampleblock}

  {\scriptsize \textbf{修正後:} テストして OK なら「コミットして」}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}{Step 6: 振り返り — CLAUDE.md を育てる}
  \begin{block}{チュートリアルで体験したワークフロー}
    {\footnotesize
    \begin{tabular}{rll}
      \hline
      \textbf{Step} & \textbf{やったこと} & \textbf{対応する機能} \\
      \hline
      1 & 環境構築 & \texttt{claude /init} で CLAUDE.md 作成 \\
      2 & 要件定義 & 目的・機能・制約を言語化 \\
      3 & 計画 & Plan Mode で設計→承認 \\
      4 & 実装 & 承認後にコード生成→テスト \\
      5 & バグ修正 & エラー報告→修正→コミット \\
      6 & 振り返り & CLAUDE.md にルールを追加 \\
      \hline
    \end{tabular}
    }
  \end{block}

  \vspace{0.2em}

  \begin{exampleblock}{次にやること: CLAUDE.md を改善する}
    {\footnotesize
    チュートリアルで気づいたことを CLAUDE.md に追記しよう: \\
    例: 「エラー処理は必ず入れること」「引数なしの場合はヘルプを表示」\\
    → 次回から Claude Code が自動的にこのルールに従う = \textbf{育てる}}
  \end{exampleblock}
\end{frame}

\begin{frame}{付録: Claude Code と Codex CLI（OpenAI）の違い}
  \begin{block}{どちらもローカルで動く CLI コーディングエージェント}
    {\small Codex CLI を使っている人向けに、主な違いを整理する。
    指示書ファイルの仕組みは共通点が多い}
  \end{block}

  \begin{table}
    \centering\tiny
    \begin{tabular}{lp{37mm}p{37mm}}
      \hline
      & \textbf{Claude Code} & \textbf{Codex CLI} \\
      \hline
      指示書ファイル &
        \textbf{CLAUDE.md} &
        \textbf{AGENTS.md}（同じ役割） \\
      階層構造 &
        グローバル → プロジェクト → サブディレクトリ &
        同様の階層構造あり\\（AGENTS.override.md で上書きも可） \\
      パス固有ルール &
        \texttt{.claude/rules/*.md} で\\パスごとにルールを分けられる &
        なし（単一ファイル） \\
      自動記憶 &
        \textbf{MEMORY.md}（AI が自動で\\学習を記録。ローカル専用） &
        \textbf{なし}（AGENTS.md を\\手動で編集する必要がある） \\
      計画モード &
        Plan Mode で設計→承認→実装 &
        なし \\
      自動化・拡張 &
        Hooks / Skills / MCP &
        限定的 \\
      権限管理 &
        suggest / auto-edit / full-auto &
        suggest / auto-edit / full-auto \\
      \hline
    \end{tabular}
  \end{table}

  {\scriptsize \textbf{最大の違い:}
    Claude Code は MEMORY.md で AI が自動的に学習を蓄積する。
    Codex CLI は AGENTS.md を人間が手動で育てる必要がある}
\end{frame}
