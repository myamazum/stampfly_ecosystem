% === P2: コーディングエージェントとは ===
\begin{frame}{コーディングエージェントとは}
  \begin{block}{定義}
    AI がターミナル上でコードを読み書きし、ビルド・テスト・Git 操作まで
    \textbf{自律的に実行}するツール
  \end{block}

  \vspace{0.3em}

  \begin{table}
    \centering\small
    \begin{tabular}{lll}
      \hline
      & \textbf{チャット AI} & \textbf{コーディングエージェント} \\
      \hline
      動作     & 対話（コピペが必要）  & ファイルを直接編集 \\
      範囲     & 1ファイルの断片      & プロジェクト全体 \\
      実行     & ユーザーが手動実行    & ビルド・テストも自律実行 \\
      Git      & 関与しない           & コミット・PR 作成まで \\
      \hline
    \end{tabular}
  \end{table}

  \vspace{0.3em}

  {\footnotesize \textbf{なぜ CLI なのか:}
    IDE プラグインと違い、SSH 先・Docker 内・CI 上など\textbf{どこでも動く}}
\end{frame}

% ------------------------------------------------------------------
% === P3: Claude Code とは ===
\begin{frame}{Claude Code とは}
  \begin{block}{概要}
    Anthropic 公式の CLI エージェント。ターミナルから Claude を呼び出し、
    コード読み書き・ビルド・テスト・Git 操作を自律的に実行する
  \end{block}

  \vspace{0.3em}

  \begin{table}
    \centering\small
    \begin{tabular}{ll}
      \hline
      \textbf{機能} & \textbf{説明} \\
      \hline
      CLAUDE.md       & プロジェクト固有の指示書（常に読み込まれる） \\
      Memory          & セッション間で記憶を永続化（4タイプ） \\
      Plan Mode       & Shift+Tab で読み取り専用の設計モード \\
      Subagent        & 並列タスク・重い処理を子プロセスに委任 \\
      Hooks \& Skills & 自動化とカスタムワークフロー \\
      Git 統合        & コミット・PR 作成・ブランチ操作 \\
      \hline
    \end{tabular}
  \end{table}

  {\scriptsize
    インストール: \texttt{npm install -g @anthropic-ai/claude-code} \quad
    起動: \texttt{cd your-project \&\& claude}
  }
\end{frame}

% === P4: なぜ「計画」が必要か ===
\begin{frame}{なぜ「計画」が必要か}
  \begin{alertblock}{AI に自由にコードを書かせると何が起きるか}
    \begin{itemize}\small
      \item 意図と違う設計で大量のファイルを変更 → \warn{手戻りコスト大}
      \item テストなしで実装を進める → バグの発見が遅れる
      \item 既存コードの意図を誤解 → 動いていた機能が壊れる
    \end{itemize}
  \end{alertblock}

  \vspace{0.5em}

  \begin{exampleblock}{解決策: 「承認」ステップを挟む}
    {\small
    \textbf{1.} AI に計画を出させる（コードは書かせない）\\
    \textbf{2.} 人間が計画をレビューして承認/修正 \\
    \textbf{3.} 承認後に初めて実装を開始
    }
  \end{exampleblock}

  \vspace{0.3em}

  {\footnotesize \textbf{原則:} 「考える」と「書く」を分離する。これが手戻りを防ぐ最大のコツ}
\end{frame}

% ------------------------------------------------------------------
% === P5: Plan Mode — 設計してから書く ===
\begin{frame}{Plan Mode — 設計してから書く}
  \begin{block}{Plan Mode とは}
    \textbf{Shift+Tab} でトグル。コード編集を\textbf{できない}読み取り専用モード。\\
    調査・設計に集中させ、方針が固まってから実装に移る
  \end{block}

  \vspace{0.2em}

  \begin{columns}[T]
    \begin{column}{0.45\textwidth}
      \begin{alertblock}{有効な場面}
        \begin{itemize}\small
          \item 複数ファイルにまたがる変更
          \item アーキテクチャの判断が必要
          \item 失敗コストが高い作業
          \item 初めて触るコードベース
        \end{itemize}
      \end{alertblock}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{exampleblock}{Plan Mode でできること}
        {\small
        ファイル読み取り・検索 \\
        コードベースの構造分析 \\
        変更計画の策定・提案 \\
        影響範囲の調査 \\[4pt]
        \warn{$\times$} ファイル編集・作成 \\
        \warn{$\times$} コマンド実行
        }
      \end{exampleblock}
    \end{column}
  \end{columns}

  \vspace{0.3em}

  {\footnotesize \textbf{Tips:} 大きな変更ほど Plan Mode が効く。5分の計画で1時間の手戻りを防げる}
\end{frame}

% === P8: なぜ Git が重要か ===
\begin{frame}{なぜ Git が重要か}
  \begin{alertblock}{AI がファイルを自律的に変更する世界では}
    {\small
    変更履歴がなければ「何が変わったか」「なぜ変わったか」が追えない。\\
    \warn{Git なしで AI にコードを書かせるのは、セーブなしでゲームをするようなもの}
    }
  \end{alertblock}

  \vspace{0.3em}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Git が提供する安全網}
        \begin{itemize}\small
          \item 変更の差分を確認できる
          \item いつでも前の状態に戻せる
          \item 誰が（AI/人間）何を変えたか記録
          \item ブランチで安全に実験できる
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Claude Code の Git 統合}
        \begin{itemize}\small
          \item 変更後に自動コミット提案
          \item Conventional Commits 準拠
          \item PR 作成・差分要約
          \item コミット履歴から文脈を読む
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}

  \vspace{0.3em}

  {\footnotesize \textbf{原則:} AI と協働するなら Git は必須。「こまめにコミット」が最大の保険}
\end{frame}

% === P11: ブランチ戦略と安全運用 ===
\begin{frame}{ブランチ戦略と安全運用}
  \begin{alertblock}{main ブランチを直接変更しない}
    {\small AI が \texttt{git push origin main} を実行すると、レビューなしで本番に反映される。
    必ず feature branch → PR → merge のフローを守る}
  \end{alertblock}

  \vspace{0.3em}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{推奨フロー}
        {\scriptsize
        \textbf{1.} \texttt{git checkout -b feat/auth} \\
        \textbf{2.} 実装・テスト・コミット \\
        \textbf{3.} \texttt{gh pr create} で PR 作成 \\
        \textbf{4.} レビュー・承認後にマージ \\
        \textbf{5.} ブランチを削除
        }
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{権限制限で安全に}
        {\scriptsize
        \texttt{settings.json} で制御:
        \begin{itemize}
          \item \texttt{permissions.deny} で\\
                \texttt{git push origin main} をブロック
          \item \texttt{--allowedTools} で\\
                使えるツールを制限
          \item GitHub の branch protection\\
                ルールと組み合わせる
        \end{itemize}
        }
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.3em}

  {\footnotesize \textbf{原則:} AI に「何をさせるか」だけでなく「何をさせないか」も設定する}
\end{frame}

\begin{frame}{CLAUDE.md — プロジェクトの知識を蓄積する}
  \begin{block}{階層構造（すべて同時に読み込まれる）}
    {\small
    \begin{tabular}{lp{60mm}}
      \api{\textasciitilde/.claude/CLAUDE.md} & グローバル設定（全プロジェクト共通）\\
      \api{<repo>/CLAUDE.md}      & プロジェクト設定（Git 管理） \\
      \api{<subdir>/CLAUDE.md}    & ディレクトリ固有の設定 \\
      \api{.claude/rules/*.md}    & パス限定ルール（paths フロントマター）\\
    \end{tabular}
    }
  \end{block}

  \vspace{0.2em}

  \begin{alertblock}{書くべき内容}
    \begin{itemize}\footnotesize
      \item ビルド・テストコマンド / コーディング規約 / コミット形式
      \item \warn{やってはいけないこと}（アンチパターン）← 最も効果が高い
      \item プロジェクト構造の概要
    \end{itemize}
  \end{alertblock}

  \vspace{0.2em}

  {\footnotesize \textbf{重要:} CLAUDE.md を書く = \textbf{AI に知識を教える}。
    ドキュメント化のコストが「AI の教育投資」として返ってくる}
\end{frame}

% === P14: Claude Code にとっての「知識」とは ===
\begin{frame}{Claude Code にとっての「知識」とは}
  \begin{block}{Claude Code は毎回まっさらな状態で起動する}
    {\small
    前回の会話は覚えていない。ではどうやって「このプロジェクトのルール」を知るのか？\\
    → \textbf{ファイルに書かれた情報を毎回読み込む}ことで知識を得る}
  \end{block}

  \vspace{0.3em}

  \begin{table}
    \centering\small
    \begin{tabular}{lp{75mm}}
      \hline
      \textbf{知識の種類} & \textbf{具体例} \\
      \hline
      プロジェクトのルール &
        「テストは pytest で実行」「コミットは英語で書く」 \\

% === P17: Hooks とは — イベント駆動の自動化 ===
\begin{frame}{Hooks とは — イベント駆動の自動化}
  \begin{block}{概念}
    {\small Claude がツールを使う前後に、シェルコマンドを\textbf{自動実行}する仕組み。\\
    「ファイルを保存したら自動で lint を走らせる」のようなことができる}
  \end{block}

  \vspace{0.2em}

  \begin{table}
    \centering\small
    \begin{tabular}{lp{65mm}}
      \hline
      \textbf{イベント} & \textbf{用途} \\
      \hline
      PreToolUse  & ツール実行前。危険な操作をブロック \\
      PostToolUse & ツール実行後。自動 lint / format \\
      Notification & 通知発生時。Slack / メール転送 \\
      Stop        & 応答完了後。自動テスト実行 \\
      \hline
    \end{tabular}
  \end{table}

  \vspace{0.2em}

  {\footnotesize
    設定場所: \api{.claude/settings.json}（Git管理）
    または \api{settings.local.json}（ローカル専用）\\
    prompt-based（テキスト判定）/ agent-based（AI判定）も選択可能}
\end{frame}

% ------------------------------------------------------------------
% === P18: Hooks の作り方と活用例 ===
\begin{frame}[fragile]{Hooks の作り方と活用例}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{例1: 自動フォーマット}
        \begin{lstlisting}[language={}, numbers=none]
{
  "hooks": {
    "PostToolUse": [{
      "matcher": "Write|Edit",
      "command": "npx prettier
        --write $TOOL_INPUT_PATH"
    }]
  }
}
        \end{lstlisting}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{例2: 危険コマンドブロック}
        \begin{lstlisting}[language={}, numbers=none]
{
  "hooks": {
    "PreToolUse": [{
      "matcher": "Bash",
      "command": "if echo
        $TOOL_INPUT | grep -q
        'rm -rf'; then
        exit 2; fi"
    }]
  }
}
        \end{lstlisting}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.3em}

  \begin{exampleblock}{その他の活用例}
    {\scriptsize
    \textbf{Notification Hook:} 長時間タスク完了時にデスクトップ通知を送る \\
    \textbf{Stop Hook:} Claude の応答完了後に自動でテストスイートを実行
    }
  \end{exampleblock}
\end{frame}

% === P22: 初心者がやりがちな失敗 ===
\begin{frame}{初心者がやりがちな失敗}
  \begin{table}
    \centering\footnotesize
    \begin{tabular}{p{40mm}p{55mm}}
      \hline
      \textbf{失敗パターン} & \textbf{何が起きるか} \\
      \hline
      Plan なしでいきなり実装させる &
        意図と違う設計で大量変更 → やり直し \\
      Git を使わない &
        AI の変更を追えない → 戻せない \\
      CLAUDE.md を書かない &
        毎回同じ指示を繰り返す → 非効率 \\
      main に直接 push &
        レビューなしで本番反映 → 障害リスク \\
      Hooks を使わず手作業を繰り返す &
        lint 忘れ・フォーマット崩れが頻発 \\
      Memory を活用しない &
        同じミスを何度も指摘する必要がある \\
      \hline
    \end{tabular}
  \end{table}

  {\scriptsize \textbf{共通の原因:}
    AI を「魔法の箱」として扱ってしまう →
    \textbf{対策:} ガードレール（Plan, Git, CLAUDE.md）を先に整える}
\end{frame}

% === P23: Top 10 Tips ===
\begin{frame}{Top 10 Tips}
  \begin{enumerate}\small
    \item \textbf{Plan Mode で考えてから書く} — 5分の計画で1時間の手戻りを防ぐ
    \item \textbf{Git は必須} — こまめにコミット、Next steps を含める
    \item \textbf{feature branch → PR → merge} — main を直接触らない
    \item \textbf{CLAUDE.md は AI への教育投資} — 書くほど賢くなる
    \item \textbf{ガードレール $>$ 指示} — 「やるな」の方が「やれ」より効果的
    \item \textbf{feedback メモリで学習} — 同じミスを二度と繰り返さない
    \item \textbf{Claude Code を育てる} — 失敗するたびにルールを追加
    \item \textbf{Hooks で自動化} — 手作業の繰り返しを排除
    \item \textbf{Skills でワークフロー化} — 複雑な手順を1コマンドに
    \item \textbf{GitHub = AI の知識のハブ} — clone すれば知識も一緒に来る
  \end{enumerate}
\end{frame}
