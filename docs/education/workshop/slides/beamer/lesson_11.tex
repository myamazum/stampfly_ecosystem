\begin{frame}[fragile]{コード例: 基本フライト / Basic Flight}
  \begin{lstlisting}[style=sfpython, title={flight\_basic.py}]
from stampfly import StampFly

drone = StampFly()
drone.connect()

drone.takeoff()
drone.move_forward(50)    # 50 cm forward
drone.rotate_clockwise(90)
drone.move_forward(50)
drone.land()

drone.end()
  \end{lstlisting}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{コード例: テレメトリ取得 / Telemetry}
  \begin{lstlisting}[style=sfpython, title={flight\_telemetry.py}]
import time
from stampfly import StampFly

drone = StampFly()
drone.connect()
drone.takeoff()

# Hover + read telemetry for 5 seconds
for _ in range(50):
    drone.send_rc_control(0, 0, 0, 0)
    t = drone.get_telemetry()
    print(f"alt={t.get('alt',0):.1f}")
    time.sleep(0.1)

drone.land()
drone.end()
  \end{lstlisting}
\end{frame}
