% ------------------------------------------------------------------
% Slide 5: Reconstructing Plant I/O
% ------------------------------------------------------------------
\begin{frame}{プラント入出力の復元 / Reconstructing Plant I/O}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{テレメトリの記録内容}
        \begin{tabular}{ll}
          \code{ctrl\_roll} & スティック入力 \\
          \code{gyro\_x} & 角速度（計測値） \\
        \end{tabular}
      \end{block}

      \vspace{0.5em}

      \begin{exampleblock}{復元手順（$K_p$ 既知）}
        {\footnotesize
        \begin{enumerate}
          \item $\text{target} = \text{ctrl} \times \text{rate\_max}$
          \item $u(t) = K_p \!\times\! (\text{target} - \text{gyro})$
          \item $y(t) = \text{gyro}$
        \end{enumerate}
        }
      \end{exampleblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{なぜ開ループ同定が可能?}
        {\small
        $K_p$ が既知なら、プラントへの入力 $u(t)$ を計算できる。
        閉ループデータでも、開ループモデルを直接フィッティングできる。
        }
      \end{alertblock}

      \vspace{0.5em}

      {\footnotesize
      \begin{tabular}{ll}
        \textbf{パラメータ} & \textbf{値} \\
        \hline
        $K_p$ & L5 の値（例: 0.5） \\
        \code{rate\_max} & 1.0\,rad/s \\
      \end{tabular}
      }
    \end{column}
  \end{columns}
\end{frame}

% ------------------------------------------------------------------
% Slide 7: sf sysid fit
% ------------------------------------------------------------------
\begin{frame}[fragile]{sf sysid fit / Model Fitting Tool}
  \begin{block}{コマンド}
    \code{sf sysid fit flight.csv -{}-kp 0.5 -{}-plot}
  \end{block}

  \vspace{0.3em}

  \begin{lstlisting}[style=sfcpp, title={出力例}]
=== Plant Model Identification ===
  Model: G_p(s) = K / (s * (tau_m * s + 1))

  Roll   K =  98.5 (ref: 102.0, err: 3.4%)
         tau_m = 0.019 (ref: 0.020, err: 5.0%)  R2 = 0.94
  Pitch  K =  72.1 (ref:  70.0, err: 3.0%)
         tau_m = 0.021 (ref: 0.020, err: 5.0%)  R2 = 0.91
  \end{lstlisting}

  \vspace{0.3em}

  {\small
  \begin{tabular}{ll}
    \code{-{}-kp 0.5} & 飛行時の $K_p$（\textbf{必須}） \\
    \code{-{}-axis roll} & 特定軸のみ分析 \\
    \code{-{}-rate-max 1.0} & \code{rate\_max}（roll/pitch 既定=1.0, yaw=5.0） \\
    \code{-o result.yaml} & 結果をファイルに保存 \\
  \end{tabular}
  }
\end{frame}

\begin{frame}{モデル検証 / Model Verification}
  \begin{table}
    \centering
    \begin{tabular}{lccccc}
      \hline
      \textbf{軸} & \textbf{$K$ (同定)} & \textbf{$K$ (L6理論)} & \textbf{$\tau_m$ (同定)} & \textbf{$\tau_m$ (理論)} & \textbf{$R^2$} \\
      \hline
      Roll  & ? & 102.0 & ? & 0.020 & ? \\
      Pitch & ? &  70.0 & ? & 0.020 & ? \\
      Yaw   & ? &  19.0 & ? & 0.020 & ? \\
      \hline
    \end{tabular}
  \end{table}

  \vspace{0.5em}

  \begin{block}{設計 $K_p$ の計算（$\zeta = 0.7$）}
    \vspace{-0.5em}
    \[
      K_p = \frac{1}{4\,\zeta^2\,K\,\tau_m}
    \]
    \vspace{-1em}
  \end{block}

  \vspace{0.3em}

  \begin{exampleblock}{考察}
    {\small
    \begin{itemize}
      \item 同定値と理論値のずれ → モデル誤差・無駄時間が原因
      \item 同定 $K$, $\tau_m$ で設計した $K_p$ は L6 の理論値と近いか?
    \end{itemize}
    }
  \end{exampleblock}
\end{frame}
