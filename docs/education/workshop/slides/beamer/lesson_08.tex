      \hline
      $L$ & 0.023\,m & アーム長（中心$\to$モータ）\\
      $k_q$ & 0.01 & トルク対推力比 \\
      \hline
    \end{tabular}
  \end{table}

  \vspace{0.5em}

  \begin{block}{ミキサー方程式}
    \begin{itemize}
      \item $M_1 = T/4 + \tau_\phi/(4L) - \tau_\theta/(4L) - \tau_\psi/(4k_q)$
      \item $M_2 = T/4 + \tau_\phi/(4L) + \tau_\theta/(4L) + \tau_\psi/(4k_q)$
      \item $M_3 = T/4 - \tau_\phi/(4L) + \tau_\theta/(4L) - \tau_\psi/(4k_q)$
      \item $M_4 = T/4 - \tau_\phi/(4L) - \tau_\theta/(4L) + \tau_\psi/(4k_q)$
    \end{itemize}
  \end{block}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{実習: カスタムミキサー / Hands-on}
  \begin{lstlisting}[style=sfcpp, title={student.cpp (excerpt)}]
// Custom mixer from first principles
float L = 0.023f, kq = 0.01f;
float T = throttle;
float R = roll_cmd, P = pitch_cmd, Y = yaw_cmd;

float m1 = T/4 + R/(4*L) - P/(4*L) - Y/(4*kq);
float m2 = T/4 + R/(4*L) + P/(4*L) + Y/(4*kq);
float m3 = T/4 - R/(4*L) + P/(4*L) - Y/(4*kq);
float m4 = T/4 - R/(4*L) - P/(4*L) + Y/(4*kq);

// Clamp [0.0, 1.0]
if (m1 < 0) m1 = 0; if (m1 > 1) m1 = 1;
if (m2 < 0) m2 = 0; if (m2 > 1) m2 = 1;
if (m3 < 0) m3 = 0; if (m3 > 1) m3 = 1;
if (m4 < 0) m4 = 0; if (m4 > 1) m4 = 1;

ws::motor_set_duty(1, m1); ws::motor_set_duty(2, m2);
ws::motor_set_duty(3, m3); ws::motor_set_duty(4, m4);
  \end{lstlisting}
\end{frame}
