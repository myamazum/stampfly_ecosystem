    I 項・D 項を追加し、定常偏差除去と\\オーバーシュート低減を実現する
  \end{block}

  \vspace{1em}

  \begin{itemize}
    \item P 制御の限界（定常偏差が残る）
    \item I 項: 偏差の積分で定常偏差を除去
    \item D 項: 偏差の微分でオーバーシュートを低減
  \end{itemize}

  \vspace{0.5em}

  \begin{block}{チューニングの順序}
    \begin{enumerate}
      \item $K_i = K_d = 0$ にして $K_p$ を調整（振動しない最大値）
      \item $K_i$ を少しずつ追加（定常偏差が消えるまで）
      \item $K_d$ を少しずつ追加（振動が収まるまで）
    \end{enumerate}
  \end{block}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{実習: ロール軸 PID / Hands-on}
  \begin{lstlisting}[style=sfcpp, title={student.cpp (roll axis excerpt)}]
float Kp=0.5f, Ki=0.3f, Kd=0.005f;
float integral=0, prev_err=0;
// ... inside loop_400Hz(dt) ...
float target = ws::rc_roll() * 1.0f;
float error  = target - ws::gyro_x();
float P = Kp * error;
integral += error * dt;
if (integral >  0.5f) integral =  0.5f; // Clamp
if (integral < -0.5f) integral = -0.5f;
float I = Ki * integral;
float D = Kd * (error - prev_err) / dt;
prev_err = error;
float roll_out = P + I + D;
// Repeat for pitch and yaw axes
ws::motor_mixer(ws::rc_throttle(), roll_out,
                pitch_out, yaw_out);
  \end{lstlisting}
\end{frame}
