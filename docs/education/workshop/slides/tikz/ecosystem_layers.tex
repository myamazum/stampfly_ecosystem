\definecolor{cfirmware}{RGB}{0, 100, 180}
\definecolor{ccontrol}{RGB}{40, 140, 60}
\definecolor{canalysis}{RGB}{180, 120, 0}
\definecolor{csimulator}{RGB}{120, 80, 160}
\definecolor{ctools}{RGB}{100, 100, 100}
\definecolor{cprotocol}{RGB}{180, 60, 40}

\begin{tikzpicture}[
    box/.style={
      rectangle, rounded corners=5pt, draw=#1, line width=1.5pt,
      fill=#1!10, minimum width=42mm, minimum height=16mm,
      font=\bfseries\large, text=#1, align=center
    },
    infra/.style={
      rectangle, rounded corners=3pt, draw=#1, line width=1pt,
      fill=#1!8, minimum width=28mm, minimum height=10mm,
      font=\bfseries\normalsize, text=#1, align=center
    },
    desc/.style={
      font=\small, text=black!60, align=center
    },
    arr/.style={
      -{Stealth[length=3.5mm]}, line width=1.5pt, #1
    },
    arrlabel/.style={
      font=\scriptsize, text=black!55, align=center, fill=white,
      inner sep=1.5pt
    },
  ]

  % === Main cycle nodes ===
  % Top: control/
  \node[box=ccontrol] (ctrl) at (0, 4.2) {\texttt{control/}\\[-2pt]{\normalsize 制御を設計}};

  % Center: firmware/
  \node[box=cfirmware] (fw) at (0, 0) {\texttt{firmware/}\\[-2pt]{\normalsize 実装して飛ばす}};

  % Left: simulator/
  \node[box=csimulator] (sim) at (-6.8, 0) {\texttt{simulator/}\\[-2pt]{\normalsize 安全に検証}};

  % Right: analysis/
  \node[box=canalysis] (ana) at (6.8, 0) {\texttt{analysis/}\\[-2pt]{\normalsize データ解析}};

  % === Cycle arrows ===
  % control → firmware (design parameters flow down)
  % 制御パラメータを実装に渡す
  \draw[arr=ccontrol] (ctrl.south) -- (fw.north)
    node[arrlabel, midway, right=4mm] {PIDパラメータ};

  % firmware → simulator (test before flight)
  % 実装をシミュレータで事前テスト
  \draw[arr=cfirmware] (fw.west) -- (sim.east)
    node[arrlabel, midway, above=2mm] {仮想テスト};

  % simulator → control (refine design)
  % シミュレーション結果をもとに制御を改善
  \draw[arr=csimulator] (sim.north) -- (ctrl.west)
    node[arrlabel, pos=0.45, above left=2mm and 4mm, align=center] {設計を\\改善};

  % firmware → analysis (telemetry data)
  % フライトデータを解析に送る
  \draw[arr=cfirmware] (fw.east) -- (ana.west)
    node[arrlabel, midway, above=2mm] {テレメトリ};

  % analysis → control (insights feed back)
  % 解析結果から制御を改善
  \draw[arr=canalysis] (ana.north) -- (ctrl.east)
    node[arrlabel, pos=0.45, above right=2mm and 4mm, align=center] {改善点を\\フィードバック};

  % === Infrastructure (below main cycle) ===
  \node[infra=ctools] (tools) at (-2.5, -2.8) {\texttt{tools/}\\[-2pt]{\scriptsize sf CLI}};
  \node[infra=cprotocol] (proto) at (2.5, -2.8) {\texttt{protocol/}\\[-2pt]{\scriptsize 通信仕様 SSOT}};

  % Dashed lines to show infrastructure supports everything
  % ダッシュ線でインフラが全体を支えることを示す
  \draw[dashed, line width=0.8pt, ctools!60]
    (tools.north) -- (fw.south west);
  \draw[dashed, line width=0.8pt, cprotocol!60]
    (proto.north) -- (fw.south east);

  % Infrastructure label
  % インフラ層のラベル
  \node[font=\small\itshape, text=black!45] at (0, -3.8) {全体を支えるインフラ};

\end{tikzpicture}
