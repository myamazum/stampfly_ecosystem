\documentclass[border=10pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta, positioning, fit, decorations.pathmorphing}

\begin{document}

\definecolor{ctrlcolor}{RGB}{0, 120, 200}
\definecolor{espcolor}{RGB}{120, 80, 180}
\definecolor{apicolor}{RGB}{50, 140, 50}
\definecolor{gray1}{RGB}{120, 120, 120}
\begin{tikzpicture}[
    box/.style={
      rectangle, rounded corners=4pt, draw=#1, line width=1.5pt,
      fill=#1!8, minimum width=40mm, minimum height=18mm,
      font=\large\bfseries, text=#1, align=center
    },
    smallbox/.style={
      rectangle, rounded corners=2pt, draw=#1, line width=1pt,
      fill=#1!6, minimum width=34mm, minimum height=14mm,
      font=\bfseries, text=#1, align=center
    },
    arr/.style={-{Stealth[length=3mm]}, line width=1.5pt, #1},
    lbl/.style={font=\small, text=gray1, align=center},
  ]

  % === Top row: Controller -> ESP-NOW -> StampFly ===
  \node[box=ctrlcolor] (ctrl) at (0, 0) {Controller};
  \node[box=espcolor, right=30mm of ctrl] (espnow) {ESP-NOW\\2.4\,GHz};
  \node[box=apicolor, right=30mm of espnow] (stampfly) {StampFly};

  \draw[arr=espcolor, decorate, decoration={snake, amplitude=1.5mm, segment length=6mm}]
    (ctrl) -- node[above=3mm, font=\small\bfseries, text=espcolor] {Wireless} (espnow);
  \draw[arr=apicolor] (espnow) -- (stampfly);

  % === Controller side: 2 physical sticks ===
  \node[smallbox=ctrlcolor, below left=35mm and 5mm of ctrl]
    (lstick) {Left Stick\\[-1pt]{\footnotesize\color{gray1} Throttle / Yaw}};
  \node[smallbox=ctrlcolor, below right=35mm and 5mm of ctrl]
    (rstick) {Right Stick\\[-1pt]{\footnotesize\color{gray1} Roll / Pitch}};

  \draw[arr=ctrlcolor] (lstick) -- (ctrl);
  \draw[arr=ctrlcolor] (rstick) -- (ctrl);

  % === StampFly side: 4 API functions in 2x2 grid ===
  %
  % Coordinate calculation (positioning: below=Xmm is EDGE-TO-EDGE gap):
  %   stampfly: center (S,0), half-w=20, half-h=9 → south=(S,-9)
  %   smallbox: half-w=17, half-h=7
  %
  %   api1/api2 (below=22mm): top=-31, center=-38, bottom=-45
  %     api1 x=[S-39,S-5], api2 x=[S+5,S+39], gap at x=S (10mm)
  %   api3/api4 (below=46mm): top=-55, center=-62, bottom=-69
  %     api3 x=[S-39,S-5], api4 x=[S+5,S+39]
  %
  % Fork routing (orthogonal, no diagonals):
  %   fork1 at y=-28 (3mm above api1/api2 top=-31)
  %     horizontal bar: (S-22,-28)—(S+22,-28), all above api1/api2 ✓
  %     vertical drops: (S-22,-28)→api1.north(S-22,-31), 3mm ✓
  %                     (S+22,-28)→api2.north(S+22,-31), 3mm ✓
  %   trunk continues: (S,-28)→(S,-48) through gap at x=S ✓
  %   fork2 at y=-48 (3mm below api1/api2 bottom=-45)
  %     horizontal bar: (S-22,-48)—(S+22,-48), all below api1/api2 ✓
  %     vertical drops: (S-22,-48)→api3.north(S-22,-55), 7mm ✓
  %                     (S+22,-48)→api4.north(S+22,-55), 7mm ✓
  %
  \node[smallbox=apicolor, below=22mm of stampfly, xshift=-22mm]
    (api1) {rc\_throttle()\\[-1pt]{\footnotesize\color{gray1} 0.0\,--\,1.0}};
  \node[smallbox=apicolor, below=22mm of stampfly, xshift=22mm]
    (api2) {rc\_roll()\\[-1pt]{\footnotesize\color{gray1} $\pm$1.0}};
  \node[smallbox=apicolor, below=46mm of stampfly, xshift=-22mm]
    (api3) {rc\_pitch()\\[-1pt]{\footnotesize\color{gray1} $\pm$1.0}};
  \node[smallbox=apicolor, below=46mm of stampfly, xshift=22mm]
    (api4) {rc\_yaw()\\[-1pt]{\footnotesize\color{gray1} $\pm$1.0}};

  % Fork routing: orthogonal (horizontal bar + vertical drop) pattern
  % Fork 1: distribute to upper row (api1, api2)
  \coordinate (fork1) at ([yshift=-19mm]stampfly.south);  % y=-28
  \draw[line width=1.5pt, apicolor] (stampfly.south) -- (fork1);
  \draw[line width=1.5pt, apicolor] (fork1) -- (fork1 -| api1.north);
  \draw[line width=1.5pt, apicolor] (fork1) -- (fork1 -| api2.north);
  \draw[arr=apicolor] (fork1 -| api1.north) -- (api1.north);
  \draw[arr=apicolor] (fork1 -| api2.north) -- (api2.north);

  % Fork 2: trunk continues through gap, distribute to lower row (api3, api4)
  \coordinate (fork2) at ([yshift=-39mm]stampfly.south);  % y=-48
  \draw[line width=1.5pt, apicolor] (fork1) -- (fork2);
  \draw[line width=1.5pt, apicolor] (fork2) -- (fork2 -| api3.north);
  \draw[line width=1.5pt, apicolor] (fork2) -- (fork2 -| api4.north);
  \draw[arr=apicolor] (fork2 -| api3.north) -- (api3.north);
  \draw[arr=apicolor] (fork2 -| api4.north) -- (api4.north);

  % Junction dots at fork points
  \fill[apicolor] (fork1) circle (2.5pt);
  \fill[apicolor] (fork2) circle (2.5pt);

\end{tikzpicture}
\end{document}
