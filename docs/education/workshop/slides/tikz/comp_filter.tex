\documentclass[border=10pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta, positioning, calc}

\begin{document}

\definecolor{sfblue}{RGB}{0, 120, 200}
\definecolor{sfred}{RGB}{220, 50, 30}
\definecolor{sfgreen}{RGB}{40, 160, 40}
\definecolor{sfgray}{RGB}{80, 80, 80}
\definecolor{sflight}{RGB}{220, 240, 255}

\begin{tikzpicture}[
    node distance=1.5cm and 2.0cm,
    block/.style={draw, #1, line width=1.2pt, fill=#1!8,
                  minimum width=2.0cm, minimum height=0.9cm,
                  font=\normalsize\bfseries, rounded corners=3pt},
    block/.default=sfblue,
    gain/.style={draw, #1, line width=1.2pt, fill=#1!8,
                 circle, minimum size=0.9cm,
                 font=\normalsize\bfseries},
    gain/.default=sfblue,
    sum/.style={draw, sfblue, line width=1.2pt, fill=white,
                circle, minimum size=0.8cm,
                font=\normalsize\bfseries},
    input/.style={font=\normalsize\bfseries, sfgray},
    output/.style={font=\large\bfseries, sfblue},
    line/.style={-{Stealth[length=2.5mm]}, line width=1.2pt, sfgray},
    pathlbl/.style={font=\normalsize\bfseries},
  ]

  % --- Input nodes ---
  \node[input] (gyro_in) {Gyro $\omega$};
  \node[input, below=3.0cm of gyro_in] (accel_in) {Accel};

  % --- Gyro path (top) ---
  \node[block=sfred, right=1.5cm of gyro_in] (integrator) {$\displaystyle\int dt$};
  \node[gain=sfred, right=1.5cm of integrator] (alpha) {$\alpha$};

  % --- Accel path (bottom) ---
  \node[block=sfgreen, right=1.5cm of accel_in] (atan) {atan2};
  \node[gain=sfgreen, right=1.5cm of atan] (one_alpha) {$1\!-\!\alpha$};

  % --- Summation ---
  \node[sum] at ($(alpha)!0.5!(one_alpha) + (2.5, 0)$) (sumnode) {$+$};

  % --- Output ---
  \node[output, right=1.2cm of sumnode] (output) {$\hat{\theta}$};

  % --- Connections ---
  % Gyro path
  \draw[line, sfred] (gyro_in) -- (integrator);
  \draw[line, sfred] (integrator) -- (alpha);
  \draw[line, sfred] (alpha.east) -- ++(0.5, 0) |- (sumnode.150);

  % Accel path
  \draw[line, sfgreen] (accel_in) -- (atan);
  \draw[line, sfgreen] (atan) -- (one_alpha);
  \draw[line, sfgreen] (one_alpha.east) -- ++(0.5, 0) |- (sumnode.210);

  % Output
  \draw[line, sfblue, line width=1.5pt] (sumnode) -- (output);

  % --- Path labels ---
  \node[pathlbl, sfred, above=0.2cm of integrator] {High-pass};
  \node[pathlbl, sfgreen, below=0.2cm of atan] {Low-pass};

\end{tikzpicture}
\end{document}
