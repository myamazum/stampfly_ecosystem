\begin{frame}[fragile]{工場出荷状態に戻す / Restore Factory Firmware}
  \begin{alertblock}{注意}
    {\small ワークショップを進めると、機体・送信機のファームウェアは上書きされます。\\
    工場出荷状態に戻したい場合は、以下のコマンドを実行してください。}
  \end{alertblock}

  \vspace{-0.2em}

  \begin{block}{機体（Vehicle）を工場出荷状態に戻す}
  \begin{lstlisting}[style=sfcpp]
sf flash vehicle --legacy
  \end{lstlisting}
  \end{block}

  \vspace{-0.3em}

  \begin{block}{送信機（Controller）を工場出荷状態に戻す}
  \begin{lstlisting}[style=sfcpp]
sf flash controller --legacy
  \end{lstlisting}
  \end{block}

  \vspace{-0.2em}

  {\scriptsize
  \textbf{注意:} 事前に \code{source setup\_env.sh} で開発環境をセットアップしてください。\\
  デバイスを USB 接続した状態で実行してください。
  }
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{リポジトリの更新 / Sync Repository}
  \vspace{-0.3em}

  \begin{block}{通常の更新}
    {\small \code{git pull} でリモートの最新変更を取り込む}
  \end{block}

  \vspace{-0.3em}

  \begin{lstlisting}[style=sfcpp]
cd ~/stampfly_ecosystem
git pull
  \end{lstlisting}

  \vspace{-0.2em}

  \begin{alertblock}{\code{git pull} でエラーが出た場合}
    {\footnotesize ローカルの変更がリモートと競合すると pull が失敗する。
    その場合はローカルを破棄してリモートに強制一致させる:}
  \end{alertblock}

  \vspace{-0.3em}

  \begin{lstlisting}[style=sfcpp]
git fetch origin
git reset --hard origin/main
git clean -fd
  \end{lstlisting}

  \vspace{-0.2em}

  {\scriptsize
  \textbf{注意:} ローカルの変更・追加ファイルはすべて失われます。
  }
\end{frame}
