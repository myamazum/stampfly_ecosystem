\begin{frame}{ARM / DISARM とは / What is ARM?}
  \begin{block}{安全スイッチ / Safety Switch}
    \begin{itemize}
      \item \textbf{ARM} = モーター出力を有効化（回転可能になる）
      \item \textbf{DISARM} = モーター出力を無効化（強制停止）
      \item コードから \api{arm()} で ARM できる
    \end{itemize}
  \end{block}

  \vspace{1em}

  \begin{alertblock}{重要 / Important}
    \warn{\api{arm()} を呼ばないとモーターは回らない！\\
    \texttt{setup()} 内で最初に呼ぶこと}
  \end{alertblock}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}{モータ制御 API}
  \begin{table}
    \centering\small
    \begin{tabular}{lll}
      \hline
      \textbf{関数} & \textbf{説明} & \textbf{引数} \\
      \hline
      \api{arm()} & モーター出力を有効化 & --- \\
      \api{disarm()} & モーター出力を無効化 & --- \\
      \api{motor\_set\_duty(id, duty)} & 個別モータ設定 & id=1--4, duty=0.0--1.0 \\
      \api{motor\_set\_all(duty)} & 全モータ一括 & duty=0.0--1.0 \\
      \api{motor\_stop\_all()} & 全モータ停止 & --- \\
      \hline
    \end{tabular}
  \end{table}

  \vspace{1em}

  \begin{alertblock}{安全注意 / Safety}
    \warn{duty は 0.15 以下で！機体を手で押さえてテスト！}
  \end{alertblock}
\end{frame}

\begin{frame}[fragile]{実習: モータを順番に回す / Hands-on}
  \begin{sflisting}{user\_code.cpp}
#include "workshop_api.hpp"
static uint32_t timer = 0;

void setup() {
    ws::print("Lesson 1: Motor Control");
    ws::arm();  // Enable motor output
}
void loop_400Hz(float dt) {
    timer++;
    // 800 ticks = 2 seconds at 400Hz
    int motor_id = (timer / 800) % 4 + 1;
    ws::motor_stop_all();
    ws::motor_set_duty(motor_id, 0.1f);
}
  \end{sflisting}
\end{frame}
