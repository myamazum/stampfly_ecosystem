\documentclass[border=10pt]{standalone}
\usepackage{tikz}
\usepackage{amsmath}
\usetikzlibrary{arrows.meta, positioning, calc, decorations.pathreplacing}

\begin{document}

\definecolor{motorcolor}{RGB}{0, 120, 200}
\definecolor{propcolor}{RGB}{180, 100, 20}
\definecolor{geomcolor}{RGB}{120, 80, 150}
\definecolor{bodycolor}{RGB}{40, 160, 40}
\definecolor{arrowcolor}{RGB}{80, 80, 80}
\begin{tikzpicture}[
    block/.style={
      rectangle, draw=#1, line width=1.4pt,
      fill=#1!8, rounded corners=3pt,
      minimum width=28mm, minimum height=15mm,
      font=\bfseries\small, text=#1, align=center
    },
    arrow/.style={-{Stealth[length=2.5mm]}, line width=1.1pt, arrowcolor},
    siglbl/.style={font=\small\bfseries, text=arrowcolor},
    desclbl/.style={font=\scriptsize, text=arrowcolor},
    sublbl/.style={font=\scriptsize\itshape},
  ]

  % Input
  \node[siglbl] (input) at (-1.2, 0) {$d$};

  % Block 1: Motor
  \node[block=motorcolor] (motor) at (2.5, 0)
    {Motor\\[1pt]$\dfrac{K_v}{\tau_m s + 1}$};

  % Block 2: Propeller
  \node[block=propcolor] (prop) at (7.0, 0)
    {Propeller\\[1pt]$F = C_t \omega_m^2$};

  % Block 3: Mixer geometry
  \node[block=geomcolor] (geom) at (11.5, 0)
    {Geometry\\[1pt]$\times\!4L$ or $\times\!4\kappa$};

  % Block 4: Rigid body
  \node[block=bodycolor] (body) at (16.0, 0)
    {Rigid Body\\[1pt]$\dfrac{1}{I \cdot s}$};

  % Output
  \node[siglbl] (output) at (19.2, 0) {$\omega$};

  % Arrows with signal labels
  \draw[arrow] (input) -- (motor);
  \draw[arrow] (motor) -- node[siglbl, above] {$\omega_m$} (prop);
  \draw[arrow] (prop)  -- node[siglbl, above] {$F$} (geom);
  \draw[arrow] (geom)  -- node[siglbl, above] {$\tau$} (body);
  \draw[arrow] (body)  -- (output);

  % Description labels (below blocks)
  \node[sublbl, motorcolor, below=2mm] at (motor.south)
    {PWM $\to$ motor speed};
  \node[sublbl, propcolor, below=2mm] at (prop.south)
    {rotation $\to$ thrust};
  \node[sublbl, geomcolor, below=2mm] at (geom.south)
    {differential thrust $\to$ torque};
  \node[sublbl, bodycolor, below=2mm] at (body.south)
    {torque $\to$ angular rate};

  % Signal unit labels (above arrows, smaller)
  \node[desclbl, above=0.5mm] at (input.north) {duty};
  \node[desclbl, above=6mm] at ($(motor.east)!0.5!(prop.west)$) {[rad/s]};
  \node[desclbl, above=6mm] at ($(prop.east)!0.5!(geom.west)$) {[N]};
  \node[desclbl, above=6mm] at ($(geom.east)!0.5!(body.west)$) {[N$\cdot$m]};
  \node[desclbl, above=0.5mm] at (output.north) {[rad/s]};

  % Linearization brace and note
  \draw[decorate, decoration={brace, amplitude=5pt, mirror},
        arrowcolor, line width=0.8pt]
    ($(motor.south west)+(0,-1.2)$) -- ($(geom.south east)+(0,-1.2)$)
    node[midway, below=5pt, font=\small, arrowcolor, align=center]
    {linearize at hover: $\dfrac{K_m}{\tau_m s + 1}$\,,\quad
     $K_f = 2 C_t K_v \omega_{m,h}$\,,\quad $K_m = 4L \cdot K_f$};

\end{tikzpicture}
\end{document}
