\documentclass[border=10pt]{standalone}
\usepackage{tikz}
\usepackage{amsmath}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usepgfplotslibrary{groupplots, fillbetween}
\usetikzlibrary{arrows.meta, patterns}

\begin{document}

\definecolor{nodelaycolor}{RGB}{0, 120, 200}
\definecolor{delaycolor}{RGB}{220, 50, 30}
\definecolor{gridcolor}{RGB}{220, 220, 220}
\definecolor{labelcolor}{RGB}{80, 80, 80}
\definecolor{refcolor}{RGB}{150, 150, 150}
\definecolor{dangercolor}{RGB}{255, 230, 230}
\begin{tikzpicture}
\begin{groupplot}[
  group style={
    group size=1 by 2,
    vertical sep=6mm,
    xlabels at=edge bottom,
    xticklabels at=edge bottom,
  },
  width=10cm,
  xmin=1, xmax=500,
  xmode=log,
  grid=both,
  grid style={gridcolor, thin},
  tick label style={font=\small, labelcolor},
  label style={font=\small\bfseries, labelcolor},
  every axis plot/.append style={line width=1.5pt},
]

% ===== Magnitude plot (small reference) =====
\nextgroupplot[
  height=3cm,
  ylabel={$|L|$ [dB]},
  ymin=-40, ymax=60,
  ytick={-40, 0, 40},
  yticklabels={$-40$, $0$, $40$},
]

% Magnitude for Kp=0.5 (same with/without delay)
\addplot[labelcolor, samples=200, smooth, domain=1:500]
  {8.685890*ln(51/(x*sqrt(1+0.0004*x*x)))};

% 0 dB reference
\addplot[refcolor, dashed, line width=0.8pt] coordinates {(1,0) (500,0)};

% ωgc marker
\addplot[delaycolor, only marks, mark=*, mark size=2.5pt] coordinates {(40, 0)};

% Note: magnitude unaffected
\node[font=\scriptsize, labelcolor, align=center] at (axis cs:6, 45)
  {$|e^{-j\omega\tau_d}|\!=\!1$};
\node[font=\scriptsize, labelcolor, align=center] at (axis cs:6, 30)
  {gain unchanged};

% ===== Phase comparison plot (main) =====
\nextgroupplot[
  height=6cm,
  ylabel={$\angle L(j\omega)$ [deg]},
  xlabel={$\omega$ \; [rad/s]},
  ymin=-240, ymax=-60,
  ytick={-240,-210,-180,-150,-120,-90,-60},
]

% Danger zone: below -180°
\addplot[name path=danger_top, draw=none] coordinates {(1,-180) (500,-180)};
\addplot[name path=danger_bot, draw=none] coordinates {(1,-240) (500,-240)};
\addplot[dangercolor, forget plot]
  fill between[of=danger_top and danger_bot];
\node[font=\scriptsize, delaycolor!70, rotate=0] at (axis cs:200, -220)
  {Unstable region};

% -180 degree reference line
\addplot[refcolor, dashed, line width=0.8pt] coordinates {(1,-180) (500,-180)};

% No-delay phase: -90 - atan(τm·ω)
\addplot[nodelaycolor, line width=2pt, samples=200, smooth, domain=1:500]
  {-90 - atan(0.02*x)};

% With-delay phase: -90 - atan(τm·ω) - τd·ω·(180/π)
% τd=0.005, 0.005·180/π = 0.28648
\addplot[delaycolor, line width=2pt, dashed, samples=200, smooth, domain=1:500]
  {-90 - atan(0.02*x) - 0.28648*x};

% -90 degree reference
\addplot[refcolor, dashed, line width=0.6pt] coordinates {(1,-90) (500,-90)};

% Vertical line at ωgc=40
\draw[labelcolor, densely dotted, line width=0.6pt]
  (axis cs:40, -60) -- (axis cs:40, -240);

% Phase markers at ωgc=40
\addplot[nodelaycolor, only marks, mark=*, mark size=3pt]
  coordinates {(40, -128.66)};
\addplot[delaycolor, only marks, mark=square*, mark size=3pt]
  coordinates {(40, -140.12)};

% PM arrow: no delay (51°)
\draw[nodelaycolor, {Stealth[length=2mm]}-{Stealth[length=2mm]}, line width=1pt]
  (axis cs:40, -131) -- (axis cs:40, -177);
\node[font=\small\bfseries, nodelaycolor, left=2pt] at (axis cs:40, -153)
  {51\textdegree};

% PM arrow: with delay (40°)
\draw[delaycolor, {Stealth[length=2mm]}-{Stealth[length=2mm]}, line width=1pt]
  (axis cs:55, -142) -- (axis cs:55, -177);
\node[font=\small\bfseries, delaycolor, right=2pt] at (axis cs:55, -160)
  {40\textdegree};

% ΔPM annotation
\draw[delaycolor, -{Stealth[length=2mm]}, line width=0.8pt]
  (axis cs:48, -128.66) -- (axis cs:48, -140.12);
\node[font=\scriptsize\bfseries, delaycolor, right=1pt] at (axis cs:49, -134)
  {$\Delta$PM=$-$11\textdegree};

% Legend
\node[font=\small\bfseries, nodelaycolor] at (axis cs:3, -68)
  {No delay};
\node[font=\small\bfseries, delaycolor] at (axis cs:3, -78)
  {$\tau_d\!=\!5$\,ms};

\end{groupplot}
\end{tikzpicture}
\end{document}
