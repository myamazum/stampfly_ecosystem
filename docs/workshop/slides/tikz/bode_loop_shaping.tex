\documentclass[border=10pt]{standalone}
\usepackage{tikz}
\usepackage{amsmath}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usepgfplotslibrary{groupplots}
\usetikzlibrary{arrows.meta}

\begin{document}

\definecolor{kp05color}{RGB}{220, 50, 30}
\definecolor{kp025color}{RGB}{0, 120, 200}
\definecolor{gridcolor}{RGB}{220, 220, 220}
\definecolor{labelcolor}{RGB}{80, 80, 80}
\definecolor{refcolor}{RGB}{150, 150, 150}
\begin{tikzpicture}
\begin{groupplot}[
  group style={
    group size=1 by 2,
    vertical sep=8mm,
    xlabels at=edge bottom,
    xticklabels at=edge bottom,
  },
  width=10cm,
  xmin=1, xmax=1000,
  xmode=log,
  grid=both,
  grid style={gridcolor, thin},
  tick label style={font=\small, labelcolor},
  label style={font=\small\bfseries, labelcolor},
  every axis plot/.append style={line width=1.5pt},
  legend style={font=\small, fill=white, fill opacity=0.8,
                draw=gridcolor, rounded corners=2pt},
]

% ===== Magnitude plot =====
\nextgroupplot[
  height=5cm,
  ylabel={$|L(j\omega)|$ [dB]},
  ymin=-40, ymax=60,
  ytick={-40,-20,0,20,40,60},
]

% Kp=0.5 (KpK=51) magnitude curve
\addplot[kp05color, samples=200, smooth, domain=1:1000]
  {8.685890*ln(51/(x*sqrt(1+0.0004*x*x)))};

% Kp=0.25 (KpK=25.5) magnitude curve
\addplot[kp025color, samples=200, smooth, domain=1:1000]
  {8.685890*ln(25.5/(x*sqrt(1+0.0004*x*x)))};

% 0 dB reference line
\addplot[refcolor, dashed, line width=0.8pt] coordinates {(1,0) (1000,0)};

% Gain crossover markers
\addplot[kp05color, only marks, mark=*, mark size=3pt] coordinates {(40, 0)};
\addplot[kp025color, only marks, mark=*, mark size=3pt] coordinates {(23, 0)};

% ωgc labels
\node[font=\scriptsize\bfseries, kp05color, above right=1pt] at (axis cs:42, 0)
  {$\omega_{gc}\!=\!40$};
\node[font=\scriptsize\bfseries, kp025color, below left=1pt] at (axis cs:22, 0)
  {$\omega_{gc}\!=\!23$};

% Vertical dotted lines at ωgc
\draw[kp05color, densely dotted, line width=0.6pt]
  (axis cs:40, 0) -- (axis cs:40, -40);
\draw[kp025color, densely dotted, line width=0.6pt]
  (axis cs:23, 0) -- (axis cs:23, -40);

% Asymptotic slope annotations
\node[font=\scriptsize, labelcolor, rotate=-17] at (axis cs:5, 32)
  {$-20$ dB/dec};
\node[font=\scriptsize, labelcolor, rotate=-32] at (axis cs:350, -18)
  {$-40$ dB/dec};

% Break frequency
\draw[refcolor, densely dashed, line width=0.5pt]
  (axis cs:50, -40) -- (axis cs:50, 55);
\node[font=\scriptsize, refcolor] at (axis cs:50, 58)
  {$\frac{1}{\tau_m}\!=\!50$};

% Legend
\node[font=\small\bfseries, kp05color] at (axis cs:4, 52) {$K_p=0.5$};
\node[font=\small\bfseries, kp025color] at (axis cs:4, 43) {$K_p=0.25$};

% ===== Phase plot =====
\nextgroupplot[
  height=5cm,
  ylabel={$\angle L(j\omega)$ [deg]},
  xlabel={$\omega$ \; [rad/s]},
  ymin=-180, ymax=-60,
  ytick={-180,-150,-120,-90,-60},
]

% Phase curve (Kp-independent)
\addplot[labelcolor, line width=2pt, samples=200, smooth, domain=1:1000]
  {-90 - atan(0.02*x)};

% -180 degree reference line
\addplot[refcolor, dashed, line width=0.8pt] coordinates {(1,-180) (1000,-180)};

% -90 degree reference line
\addplot[refcolor, dashed, line width=0.8pt] coordinates {(1,-90) (1000,-90)};

% Vertical dotted lines at ωgc
\draw[kp05color, densely dotted, line width=0.6pt]
  (axis cs:40, -60) -- (axis cs:40, -180);
\draw[kp025color, densely dotted, line width=0.6pt]
  (axis cs:23, -60) -- (axis cs:23, -180);

% Phase at ωgc markers
\addplot[kp05color, only marks, mark=*, mark size=3pt]
  coordinates {(40, -128.66)};
\addplot[kp025color, only marks, mark=*, mark size=3pt]
  coordinates {(23, -114.7)};

% PM arrow for Kp=0.5: from phase to -180°
\draw[kp05color, {Stealth[length=2mm]}-{Stealth[length=2mm]}, line width=1pt]
  (axis cs:40, -131) -- (axis cs:40, -177);
\node[font=\small\bfseries, kp05color, right=2pt] at (axis cs:40, -155)
  {PM\!=\!51\textdegree};

% PM arrow for Kp=0.25: from phase to -180°
\draw[kp025color, {Stealth[length=2mm]}-{Stealth[length=2mm]}, line width=1pt]
  (axis cs:23, -117) -- (axis cs:23, -177);
\node[font=\small\bfseries, kp025color, left=2pt] at (axis cs:23, -148)
  {PM\!=\!65\textdegree};

\end{groupplot}
\end{tikzpicture}
\end{document}
